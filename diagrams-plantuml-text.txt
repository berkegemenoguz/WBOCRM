3.5 Activity Diagram:

flowchart TD
    Start([Start]) --> Step1[Sales Rep opens Lead Management form]
    Step1 --> Step2[Enter contact details and engagement metrics]
    Step2 --> Step3[Submit Lead Data]
    Step3 --> Decision{Is Email Unique?}
    
    Decision -- No (Scenario 3) --> Error[Display duplicate email error]
    Error --> End1([End])
    
    Decision -- Yes (Scenario 1) --> Step4[Trigger Lead Scoring Algorithm]
    Step4 --> Step5[Calculate Dynamic Priority Score]
    Step5 --> Step6[(Save to PostgreSQL Database)]
    Step6 --> Step7[Update Centralized Priority Dashboard]
    Step7 --> End2([End])

3.6 Sequence Diagram:

sequenceDiagram
    autonumber
    actor SalesRep as Sales Representative
    participant UI as React Frontend
    participant API as Spring Boot API
    participant Algo as Scoring Algorithm
    participant DB as PostgreSQL DB

    SalesRep->>UI: Submit new lead form
    UI->>API: POST request with lead data
    API->>DB: Query: checkUniqueEmail(email)
    
    alt Email exists (Scenario 3)
        DB-->>API: Return true (Conflict)
        API-->>UI: 400 Bad Request (Validation Error)
        UI-->>SalesRep: Display "Duplicate Email" warning
    else Email is unique (Scenario 1)
        DB-->>API: Return false (Clear to proceed)
        API->>Algo: calculateScore(engagementMetrics)
        Algo-->>API: return priorityScore
        API->>DB: saveLead(contactData, priorityScore)
        DB-->>API: Success (Record ID created)
        API-->>UI: 201 Created
        UI-->>SalesRep: Refresh and display Prioritized Dashboard
    end

@startuml

package "Presentation Layer (Client Tier)" {
    [React Web Application (SPA)] as UI
    note right of UI : Cross-platform web interface\nfor SME staff (Sales, Support).

package "Business Logic Layer (Application Tier - Spring Boot)" {
    [REST API Gateway] as API
    
    package "Core Business Modules" {
        [Security & RBAC Module] as Security
        [Lead & Sales Management] as Sales
        [Customer Support Module] as Support
        [Automated Lead Scoring Engine] as Engine
    }
}

package "Data Access Layer (Storage Tier)" {
    database "PostgreSQL Database" as DB
    note right of DB : Centralized Single Source of Truth\n(Leads, Tickets, Logs, Users).
}

cloud "Deployment Environment (Heroku Cloud)" {
    note "Hosts Application and Database" as CloudNote
}

' Data Flow and Relationships
UI <--> API : "HTTPS / JSON (REST)"
API --> Security : "Authenticates & Authorizes"
API --> Sales : "Routes Lead/Sales Data"
API --> Support : "Routes Ticket Data"

Sales <--> Engine : "Requests Dynamic Score"

Sales <--> DB : "Read/Write (JDBC)"
Support <--> DB : "Read/Write (JDBC)"
